Fibonacci

tantrums (.42AHH) — fibonacci.42AHH
tantrum int fib(int n)
{{
    if (n <= 1) {{ return (n); }}
    return (fib(n - 1) + fib(n - 2));
}}

tantrum void main()
{{
    for i in range(10)
    {{
        print(fib(i));
    }}
}}

Manual Memory Management

tantrums (.42AHH) — manual.42AHH
#autoFree false;

tantrum int* makeNum(int val)
{{
    int* p = alloc int(val);
    return (p);
}}

tantrum void main()
{{
    int* a = makeNum(42);
    int* b = makeNum(99);
    print(*a + *b);    // → 141
    free a;
    free b;
}}

Strict Static Typing

tantrums (.42AHH) — strict.42AHH
#mode static;

tantrum int add(int a, int b)
{{
    return (a + b);
}}

tantrum string greet(string name)
{{
    return ("Hello, " + name);
}}

tantrum void main()
{{
    int result = add(10, 20);
    string msg = greet("World");
    print(result);
    print(msg);
}}

Benchmarking

tantrums (.42AHH) — bench.42AHH
tantrum void main()
{{
    int start = getCurrentTime();

    int sum = 0;
    for i in range(1000000) {{ sum += i; }}

    int end = getCurrentTime();
    print("Sum:  " + sum);
    print("Time: " + toSeconds(end - start) + "s");
    print("RAM:  " + bytesToMB(getProcessMemory()) + " MB");
}}

Arena Allocation Pattern

tantrums (.42AHH) — arena.42AHH
// Intentional arena — allocate freely, OS reclaims on exit
#autoFree false;
#allowMemoryLeaks true;

tantrum void main()
{{
    int* a = alloc int(1);
    int* b = alloc int(2);
    int* c = alloc int(3);
    print(*a + *b + *c);   // → 6
    // not freed — intentional arena pattern
}}

Error Handling

tantrums (.42AHH) — errors.42AHH
tantrum int safeDivide(int a, int b)
{{
    if (b == 0) {{ throw "Division by zero!"; }}
    return (a / b);
}}

tantrum void main()
{{
    try
    {{
        int result = safeDivide(10, 0);
        print(result);
    }}
    catch (e)
    {{
        print("Error: " + e);   // → Error: Division by zero!
    }}
}}
